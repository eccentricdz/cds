### Default

Most common composition for displaying individual assets like NFTs or dApps.

```jsx live
function Example() {
  function NoopFn() {
    console.log('pressed');
  }
  const cards = [
    {
      title: 'Balancing the Air',
      description: '0.5 ETH',
      subtitle: 'Amber Vâ€™s Artwork',
      onClick: NoopFn,
      media: (
        <RemoteImage
          source="/img/nft.png"
          height={'100%'}
          style={{ objectFit: 'cover', cursor: 'pointer' }}
          width="100%"
        />
      ),
    },
  ];
  return (
    <VStack minHeight={300} padding={2} alignItems="center" justifyContent="center" gap={1}>
      {cards.map((card) => (
        <FloatingAssetCard {...card} />
      ))}
    </VStack>
  );
}
```
